cmake_minimum_required(VERSION 3.21.3)
project (tartar-sauce)

Include(FetchContent)

FetchContent_Declare(
  Catch2
  GIT_REPOSITORY https://github.com/catchorg/Catch2.git
  GIT_TAG        v3.0.0-preview3
)

FetchContent_MakeAvailable(Catch2)

file(GLOB LIB_SOURCES "lib/*.cpp")
add_library(tartar ${LIB_SOURCES})
target_compile_options(tartar PUBLIC -pedantic -Wall -Wextra -Wcast-align -Wdisabled-optimization -Wformat=2 -Winit-self -Wmissing-declarations -Wmissing-include-dirs -Woverloaded-virtual -Wredundant-decls -Wsign-promo -Werror -Wno-unused -std=c++17)

cmake_policy(PUSH)
  if(POLICY CMP0037)
    cmake_policy(SET CMP0037 OLD)
  endif()

file(GLOB TEST_SOURCES test/*.cpp)

foreach (testsourcefile ${TEST_SOURCES})
    add_executable("${testsourcefile}.out" "${testsourcefile}")
    target_link_libraries("${testsourcefile}.out" PRIVATE Catch2::Catch2WithMain PUBLIC tartar )
endforeach (testsourcefile ${TEST_SOURCES})

cmake_policy(POP)
